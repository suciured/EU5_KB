#"[GetScriptedGui('ctf_gvar_sel_country_sgui').Execute(GuiScope.SetRoot(Country.MakeScope).End)]"
ctf_gvar_sel_country_sgui = {
    scope = country

    is_shown = { always = yes }
    is_valid = { always = yes }

    effect = {
        set_global_variable = {
            name = ctf_gvar_sel_country
            value = root
        }
    }
}

ctf_fun_vset_sgui = {
	scope = none
	
    effect = {
		if = {
			limit = { has_global_variable = ctf_gvar_sel_country }
			
			global_var:ctf_gvar_sel_country = {
				change_country_name = ENG
			}
		}
    }
}
