ctf_vtgl_base_seff = {
    effect = {
		if = {
			limit = { has_global_variable = ctf_gvar_sel_country }
			
			if = {
				limit = { 
					global_var:ctf_gvar_sel_country = { has_variable = ctf_var_hegemon } 
				}
	
				global_var:ctf_gvar_sel_country = {
					remove_variable = ctf_var_hegemon
				}
			}
			else = {
				global_var:ctf_gvar_sel_country = {
					set_variable = ctf_var_hegemon
				}
			}
		}
    }
}

ctf_vtgl_bese_effect = {
    effect = {

		if = {
			limit = { 
				$target$ = { has_variable = ctf_var_hegemon } 
			}
	
			$target$ = {
				remove_variable = ctf_var_hegemon
			}
		}
		else = {
			$target$ = {
				set_variable = ctf_var_hegemon
			}
		}
		
    }
}

ctf_vtgl_trial_seff = {
	global_var:ctf_gvar_sel_country = {
		
		set_variable = ctf_var_hegemon
	
	}
}
